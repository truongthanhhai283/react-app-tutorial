# terminal command
	- ls: list of all directories
	- mdir: create a folder/directory
	- cd : change a directory
	- cd.. : navigate
	- clear: clear console
	- arrow key up/down: preview command
	
# npm basic
	- npm init: create a package .json file to list dependencies
	- npm install <package name> --save : install package locally(default) and save package to package.json
	- npm install <package name> -q : install package globally(access anywhere in machine) sudo
	- npm install <package name> --save-dev : use package only for developerment
	- npm build/ run build
